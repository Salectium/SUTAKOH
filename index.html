def decode_word(encoded_word, key): extended_key = (key * ((len(encoded_word) // len(key)) + 1))[:len(encoded_word)] min_length = len(encoded_word) decoded_word = [] for i in range(min_length): e_char = encoded_word[i] k_char = extended_key[i] e_index = ord(e_char.lower()) - ord('a') k_index = ord(k_char.lower()) - ord('a') backward_distance = (k_index - e_index) % 26 new_index = (e_index + backward_distance) % 26 new_char = chr(new_index + ord('a')) decoded_word.append(new_char) return ''.join(decoded_word) encoded_input = input("Enter the encoded word: ") key_input = input("Enter the key word: ") decoded = decode_word(encoded_input, key_input) print(f"Decoded Word: {decoded}")
